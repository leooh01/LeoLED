#include <FastLED.h>
#define NUM_LEDS 135
#define DATA_PIN 11
CRGB leds[NUM_LEDS];
void setup() { 
       FastLED.addLeds<NEOPIXEL, DATA_PIN>(leds, NUM_LEDS);
   }
void loop() { 
for (int i=0; i-135)
  leds[i]- (GRB; FFE5CC;)
  leds[(NUM_LEDES0-i)]-GRB;;
  FastLED.Show();
  fill_solid(leds[0],68); leds[i] = 0xFF8000
  delay(40);
  fill_solid(leds[0],68, (RGB(Black));
  delay(40);
  fill_solid(leds[69],135,) leds[i] = 0xFF8000
  delay(40);
  fill_solid(leds[69],135, (RGB(Black)));
  delay(40);
  fill_solid(leds[0],68); leds[i] = 0xFF8000
  delay(40)
  fill_solid(leds[0],68, (RGB(Black));
  delay(40);
  fill_solid(leds[69],135,) leds[i] = 0xFF8000
  delay(40);
  fill_solid(leds[69],135, (RGB(Black)));
  delay(40);
  fill_solid(leds[0],68); leds[i] = 0xFF8000
  delay(40);
  fill_solid(leds[0],68, (RGB(Black)));
  delay(40);
  fill_solid(leds[69],135,) leds[i] = 0xFF8000
  delay(40);
  fill_solid(leds[69],135, (RGB(Black)));
  delay(40);
  fill_rainbow&(leds[i]), 0,135,);
  delay(40)
}
}
